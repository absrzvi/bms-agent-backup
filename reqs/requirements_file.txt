# install.sh - Installation script

#!/bin/bash

echo "Installing Enhanced Document Processor v4.0..."

# Install core dependencies
pip install -r requirements.txt

# Download spaCy model
python -m spacy download en_core_web_sm

# Download NLTK data
python -c "
import nltk
nltk.download('punkt')
nltk.download('stopwords')
nltk.download('wordnet')
nltk.download('averaged_perceptron_tagger')
nltk.download('maxent_ne_chunker')
nltk.download('words')
"

echo "Core installation complete!"

# Optional: Install advanced features
read -p "Install optional advanced features? (y/n) " -n 1 -r
echo
if [[ $REPLY =~ ^[Yy]$ ]]
then
    pip install -r requirements-optional.txt
    echo "Advanced features installed!"
fi

# Optional: Install development tools
read -p "Install development tools? (y/n) " -n 1 -r
echo
if [[ $REPLY =~ ^[Yy]$ ]]
then
    pip install -r requirements-dev.txt
    echo "Development tools installed!"
fi

echo "Installation complete! Run 'python test_enhanced_processor.py' to verify."
